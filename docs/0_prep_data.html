                unique(spe$ROI))
#For the patient ID

                unique(spe$patient_id))
#For the sample ID
sample_id <- setNames(dittoColors(reps = 1)[seq_along(unique(spe$sample_id))], 
                unique(spe$sample_id))

                unique(spe$condition))
